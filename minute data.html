# ploting so many data points is a pain, lets define a function to do this for us
def plot_data(start_time, end_time):
    partial = data[(data['time']>=start_time)&(data['time']<=end_time)]
    price = np.array(list(partial['price']))
    time = np.array(list(partial['time']))
    x_label_index = np.linspace(0, len(time) - 1, 6).astype(int)
    x_label = np.array(list(time))[x_label_index]
    plt.figure(figsize=(10,5))
    plt.plot(time, price)
    plt.xticks(x_label_index, x_label, rotation='vertical')
    plt.show()

for h in hour:
    for m in minute:
        intervals.append(h+m)
        
clean_data = {}
for date in price_data:
    time = sorted(list(price_data[date]))
    size = len(time)
    cur = 0
    date_data = {}
    prev = None
    if time[0]<'0800' and time[-1]>'1630':
        for i in range(size):
            if prev:
                if cur<len(intervals)-1 and time[i]>intervals[cur] and time[i]<intervals[cur+1]:
                    date_data[intervals[cur]] = prev
                    cur += 1
                elif cur<len(intervals)-1 and time[i]>intervals[cur+1]: # jump condition
                    while cur<len(intervals)-1 and time[i]>intervals[cur+1]:
                        date_data[intervals[cur]] = prev
                        cur += 1
            prev = price_data[date][time[i]]
        # calculate the diff
        clean_date_data = {}
        fl = True
        val = None
        for time in date_data:
            if fl:
                val = date_data[time]
                fl = False
            else:
                diff = date_data[time]/val-1
                val = date_data[time]
                clean_date_data[time] = diff
            
        clean_data[date] = clean_date_data

# we now prepare training and testing data
def split_time(data_array, n_steps_in, n_steps_out):
    """
    split one day's data according to n_step_in, n_step_out
    i.e. [1,2,3,4,5,6] ->
    [1,2,3] ->[4,5]
    [2,3,4] ->[5,6]
    """
    X, y = [], []
    size = len(data_array)
    for i in range(size):
        end_ix = i + n_steps_in
        out_end_ix = end_ix + n_steps_out
        if out_end_ix > len(data_array): # check if n_steps out is too long
            break
        seq_x, seq_y = data_array[i:end_ix], data_array[end_ix:out_end_ix]
        X.append(seq_x) # feature
        y.append(seq_y) # target
    return np.array(X), np.array(y)

def generate_date(n_steps_in, n_steps_out):
    compiled_data = {}
    for date in clean_data:
        features = clean_data[date]
        time = sorted(list(clean_data[date].keys()))
        feature = []
        for t in time:
            feature.append(clean_data[date][t])
        x,y = split_time(feature, n_steps_in, n_steps_out)
        compiled_data[date] = [x,y]
    return compiled_data
